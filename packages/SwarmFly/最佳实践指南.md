### 2.2 批量处理优化

```python
# 推荐: 批量执行减少开销
class BatchExecutionOptimizer:
    """批量执行优化器"""
    
    def __init__(self, batch_size: int = 100, flush_interval: float = 1.0):
        self.batch_size = batch_size
        self.flush_interval = flush_interval
        self.pending_executions: List[ExecutionRequest] = []
    
    async def submit(self, request: ExecutionRequest):
        """提交执行请求"""
        self.pending_executions.append(request)
        
        # 满足任一条件则执行批量
        if (len(self.pending_executions) >= self.batch_size or
            self._should_flush_now()):
            await self._execute_batch()
    
    async def _execute_batch(self):
        """批量执行优化"""
        if not self.pending_executions:
            return
        
        batch = self.pending_executions[:self.batch_size]
        self.pending_executions = self.pending_executions[self.batch_size:]
        
        # 并行执行
        await asyncio.gather(*[self._execute_one(r) for r in batch])
```

## 五、监控告警实践

### 5.1 告警阈值设置

```python
# 告警阈值最佳实践 - 基于历史基线
class DynamicAlertThreshold:
    """动态告警阈值"""
    
    def calculate_baseline(self, metric_name: str, days: int = 30) -> Dict:
        """基于历史数据计算基线"""
        historical_data = self.fetch_historical(metric_name, days)
        
        return {
            "mean": statistics.mean(historical_data),
            "std": statistics.stdev(historical_data),
            "p50": statistics.median(historical_data),
            "p95": self.percentile(historical_data, 95),
            "p99": self.percentile(historical_data, 99)
        }
    
    def get_thresholds(self, metric_name: str) -> Dict:
        """获取动态告警阈值"""
        baseline = self.calculate_baseline(metric_name)
        
        return {
            # 警告: 超过均值+2倍标准差
            "warning": baseline["mean"] + 2 * baseline["std"],
            
            # 严重: 超过p99
            "critical": baseline["p99"],
            
            # 极端: 超过p99的1.5倍
            "extreme": baseline["p99"] * 1.5
        }
```

### 6.1 日志级别使用

| 级别 | 使用场景 | 示例 |
|------|---------|------|
| DEBUG | 开发调试、详细流程 | "Entering function X with params: ..." |
| INFO | 正常业务流程 | "Rule executed successfully", "User logged in" |
| WARNING | 异常但可恢复 | "Cache miss", "Retry attempt 2/3" |
| ERROR | 错误需要关注 | "Database connection failed", "API timeout" |
| CRITICAL | 系统级严重问题 | "Node down", "Data corruption detected" |

import structlog

structlog.configure(
    processors=[

        structlog.stdlib.add_logger_name,

        structlog.processors.TimeStamper(fmt="iso"),
