## 一、核心设计模式

### 1.1 分层架构模式

```
┌─────────────────────────────────────────────────────────────┐
│                     表现层 (Presentation)                    │
│         FLY-1 感知层 │ FLY-4 技能层 │ API Gateway             │
├─────────────────────────────────────────────────────────────┤
│                     业务逻辑层 (Business Logic)              │
│           FLY-2 法则层 │ FLY-3 趋势层 │ 引擎协同               │
├─────────────────────────────────────────────────────────────┤
│                     基础设施层 (Infrastructure)              │
│              FLY-5 工具层 │ 消息队列 │ 存储                    │
└─────────────────────────────────────────────────────────────┘
```

**设计原则:**
- 每一层只依赖其下层
- 上层通过接口调用下层
- 允许同层内部互相调用
- 禁止跨层直接访问

```python
class ConflictResolutionStrategy(ABC):
    """冲突解决策略抽象"""
    
    @abstractmethod
    def resolve(self, conflicts: List[RuleConflict]) -> Resolution:
        pass

class PriorityFirstStrategy(ConflictResolutionStrategy):
    """优先级优先策略"""
    def resolve(self, conflicts: List[RuleConflict]) -> Resolution:
        return sorted(conflicts, key=lambda c: c.priority, reverse=True)[0]

class LatestWinsStrategy(ConflictResolutionStrategy):
    """最新优先策略"""
    def resolve(self, conflicts: List[RuleConflict]) -> Resolution:
        return max(conflicts, key=lambda c: c.created_at)

class CustomStrategy(ConflictResolutionStrategy):
    """自定义策略"""
    def resolve(self, conflicts: List[RuleConflict]) -> Resolution:
        # 业务特定逻辑
        pass

# 策略选择器
class ConflictResolver:
    def __init__(self):
        self.strategies = {
            "priority": PriorityFirstStrategy(),
            "latest": LatestWinsStrategy(),
            "custom": CustomStrategy()
        }
    
    def resolve(self, conflicts: List[RuleConflict], strategy: str) -> Resolution:
        return self.strategies[strategy].resolve(conflicts)
```

```python
class RuleExecutionObserver(ABC):
    """规则执行观察者"""
    
    @abstractmethod
    def on_execution_start(self, event: ExecutionStartEvent):
        pass
    
    @abstractmethod
    def on_execution_complete(self, event: ExecutionCompleteEvent):
        pass
    
    @abstractmethod
    def on_execution_failed(self, event: ExecutionFailedEvent):
        pass

class MetricsObserver(RuleExecutionObserver):
    """指标收集观察者"""
    def on_execution_start(self, event):
        self.increment_counter("rule_execution_started")
    
    def on_execution_complete(self, event):
        self.record_duration("rule_execution_duration", event.duration)
    
    def on_execution_failed(self, event):
        self.increment_counter("rule_execution_failed")

class AlertObserver(RuleExecutionObserver):
    """告警观察者"""
    def on_execution_failed(self, event):
        if event.failure_rate > THRESHOLD:
            self.send_alert(event)

class RuleExecutor:
    """规则执行器 (被观察对象)"""
    
    def __init__(self):
        self.observers: List[RuleExecutionObserver] = []
    
    def add_observer(self, observer: RuleExecutionObserver):
        self.observers.append(observer)
    
    def remove_observer(self, observer: RuleExecutionObserver):
        self.observers.remove(observer)
    
    async def execute(self, rule_id: str):
        self._notify_start(rule_id)
        try:
            result = await self._do_execute(rule_id)
            self._notify_complete(rule_id, result)
        except Exception as e:
            self._notify_failed(rule_id, e)
    
    def _notify_start(self, rule_id):
        event = ExecutionStartEvent(rule_id=rule_id, timestamp=datetime.now())
        for observer in self.observers:
            observer.on_execution_start(event)
```

```python
# 插件管理器
class PluginManager:
    """插件管理器"""
    
    def __init__(self, plugin_dir: str):
        self.plugin_dir = plugin_dir
        self.plugins: Dict[str, Plugin] = {}
        self._discover_plugins()
    
    def _discover_plugins(self):
        """自动发现插件"""
        for plugin_path in Path(self.plugin_dir).glob("*.py"):
            if plugin_path.name.startswith("_"):
                continue
            plugin = self._load_plugin(plugin_path)
            self.plugins[plugin.name] = plugin
    
    def _load_plugin(self, path: Path) -> Plugin:
        """加载单个插件"""
        spec = importlib.util.spec_from_file_location(path.stem, path)
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        # 验证接口实现
        plugin_class = getattr(module, "Plugin")
        if not isinstance(plugin_class, (IDataSourcePlugin, IToolPlugin)):
            raise PluginInterfaceError(f"{path} does not implement required interface")
        
        return plugin_class()
```

```python
# 发布订阅模式: 事件总线
class EventBus:
    """事件总线"""
    
    def __init__(self):
        self.subscribers: Dict[str, List[Callable]] = defaultdict(list)
        self.middleware: List[Callable] = []
    
    def subscribe(self, event_type: str, handler: Callable):
        """订阅事件"""
        self.subscribers[event_type].append(handler)
    
    def unsubscribe(self, event_type: str, handler: Callable):
        """取消订阅"""
        self.subscribers[event_type].remove(handler)
    
    async def publish(self, event: Event):
        """发布事件"""
        # 中间件处理
        for mw in self.middleware:
            event = mw.process(event)
        
        # 分发到订阅者
        handlers = self.subscribers.get(event.type, [])
        await asyncio.gather(*[h(event) for h in handlers])

# 使用示例
event_bus = EventBus()

# 订阅规则执行事件
event_bus.subscribe("rule.executed", on_rule_executed_handler)
event_bus.subscribe("rule.executed", metrics_handler)
event_bus.subscribe("rule.executed", audit_handler)

# 发布事件
await event_bus.publish(RuleExecutedEvent(
    rule_id="rule_123",
    status="success",
    duration_ms=50
))
```

```python
# 消息队列: 可靠消息传输
class ReliableMessageQueue:
    """可靠消息队列"""
    
    def __init__(self, broker: str, config: dict):
        self.broker = broker
        self.config = config
        self.publisher = self._create_publisher()
        self.consumer = self._create_consumer()
    
    async def publish(self, topic: str, message: Message, options: PublishOptions):
        """可靠发布"""
        # 1. 消息持久化
        message.persist()
        
        # 2. 发送确认
        await self.publisher.send(topic, message, mandatory=True)
        
        # 3. 等待确认或重试
        confirmed = await self.wait_for_ack(message, timeout=options.timeout)
        
        if not confirmed:
            raise PublishFailedError(f"Message {message.id} not confirmed")
        
        return message.id
    
    async def consume(self, topic: str, handler: Callable, options: ConsumeOptions):
        """可靠消费"""
        async for message in self.consumer.subscribe(topic):
            try:
                await handler(message)
                await message.ack()
            except RetryableError:
                await message.nack(requeue=True)
            except NonRetryableError:
                await message.nack(requeue=False)
                await self.dlq.publish(message)  # 发送到死信队列
```

```python
# 断路器模式实现
class CircuitBreaker:
    """断路器"""
    
    def __init__(self, name: str, config: CircuitBreakerConfig):
        self.name = name
        self.config = config
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None
    
    async def call(self, func: Callable, *args, **kwargs):
        """执行受保护的调用"""
        if self.state == CircuitState.OPEN:
            if self._should_attempt_reset():
                self._to_half_open()
            else:
                raise CircuitOpenError(f"Circuit {self.name} is OPEN")
        
        try:
            result = await func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure()
            raise
    
    def _on_success(self):
        """成功处理"""
        self.failure_count = 0
        self.success_count += 1
        
        if self.state == CircuitState.HALF_OPEN:
            if self.success_count >= self.config.half_open_max_calls:
                self._to_closed()
    
    def _on_failure(self):
        """失败处理"""
        self.failure_count += 1
        self.last_failure_time = datetime.now()
        
        if self.state == CircuitState.HALF_OPEN:
            self._to_open()
        elif self._should_trip():
            self._to_open()
    
    def _should_trip(self) -> bool:
        """判断是否应该跳闸"""
        if self.failure_count >= self.config.failure_threshold:
            return True
        
        failure_rate = self.failure_count / self._total_calls
        if failure_rate >= self.config.failure_rate_threshold:
            return True
        
        return False
```

```python
# 重试模式实现
class RetryPolicy:
    """重试策略"""
    
    def __init__(self, config: RetryConfig):
        self.max_attempts = config.max_attempts
        self.initial_interval = config.initial_interval
        self.max_interval = config.max_interval
        self.backoff_multiplier = config.backoff_multiplier
        self.retryable_exceptions = config.retryable_exceptions
    
    def execute(self, func: Callable, *args, **kwargs):
        """执行带重试的调用"""
        last_exception = None
        interval = self.initial_interval
        
        for attempt in range(1, self.max_attempts + 1):
            try:
                return func(*args, **kwargs)
            except tuple(self.retryable_exceptions) as e:
                last_exception = e
                
                if attempt == self.max_attempts:
                    break
                
                # 指数退避
                time.sleep(interval)
                interval = min(interval * self.backoff_multiplier, self.max_interval)
        
        raise last_exception

# 预配置重试策略
RETRY_POLICIES = {
    "default": RetryPolicy(RetryConfig(
        max_attempts=3,

        backoff_multiplier=2.0

    "aggressive": RetryPolicy(RetryConfig(
        max_attempts=5,

    )),
```

### 5.1 读写分离模式

```python
# 读写分离: 规则缓存策略
class RuleCacheManager:
    """规则缓存管理器"""
    
    def __init__(self, read_cache: Cache, write_cache: Cache):
        self.read_cache = read_cache
        self.write_cache = write_cache
    
    async def get_rule(self, rule_id: str) -> Optional[Rule]:
        """读取 - 使用只读缓存"""
        # 1. 检查只读缓存
        cached = await self.read_cache.get(rule_id)
        if cached:
            return cached
        
        # 2. 查询数据库
        rule = await self.db.rules.find_one({"_id": rule_id})
        
        # 3. 更新只读缓存
        if rule:
            await self.read_cache.set(rule_id, rule, ttl=300)
        
        return rule
    
    async def update_rule(self, rule_id: str, updates: Dict):
        """写入 - 双写策略"""
        # 1. 更新主数据库
        await self.db.rules.update_one({"_id": rule_id}, {"$set": updates})
        
        # 2. 删除只读缓存（失效策略）
        await self.read_cache.delete(rule_id)
        
        # 3. 写入缓存（延迟同步）
        await self.write_cache.set(rule_id, updates, ttl=60)
        
        # 4. 后台同步到只读缓存
        asyncio.create_task(self._sync_to_read_cache(rule_id))
```

### 5.2 批量处理模式

```python
# 批量处理: 批量预测请求优化
class BatchPredictionOptimizer:
    """批量预测优化器"""
    
    def __init__(self, batch_size: int = 100, flush_interval: float = 1.0):
        self.batch_size = batch_size
        self.flush_interval = flush_interval
        self.pending: Dict[str, List[PredictionRequest]] = defaultdict(list)
        self.lock = asyncio.Lock()
    
    async def submit(self, metric: str, request: PredictionRequest) -> str:
        """提交预测请求"""
        async with self.lock:
            self.pending[metric].append(request)
            
            if len(self.pending[metric]) >= self.batch_size:
                await self._flush_metric(metric)
            else:
                asyncio.create_task(self._delayed_flush(metric))
            
            return request.request_id
    
    async def _flush_metric(self, metric: str):
        """批量执行单个指标的预测"""
        if not self.pending[metric]:
            return
        
        batch = self.pending[metric][:self.batch_size]
        self.pending[metric] = self.pending[metric][self.batch_size:]
        
        # 批量调用预测服务
        results = await self.prediction_service.batch_predict(metric, batch)
        
        # 分发结果
        for request, result in zip(batch, results):
            request.future.set_result(result)
```

#### 6.3.2 条件检查示例

```python
class TransitionConditionChecker:
    """转换条件检查器"""
    
    def __init__(self):
        self.conditions = TRANSITION_CONDITIONS
    
    async def check_conditions(
        self,
        rule: Rule,
        transition: str
    ) -> ConditionCheckResult:
        """检查所有转换条件"""
        result = ConditionCheckResult(
            transition=transition,
            rule_id=rule.id,
            passed=True,
            failed_conditions=[],
            skipped_conditions=[],
            warnings=[]
        )
        
        conditions = self.conditions.get(transition, [])
        
        for condition in sorted(conditions, key=lambda c: c.priority):
            try:
                if condition.timeout:
                    passed = await self._check_with_timeout(
                        condition.check_function,
                        rule,
                        condition.timeout
                    )
                else:
                    passed = condition.check_function(rule)
                
                if passed:
                    result.passed = result.passed and True
                else:
                    if condition.blocking:
                        result.passed = False
                        result.failed_conditions.append(condition.name)
                    else:
                        result.skipped_conditions.append(condition.name)
                        
            except Exception as e:
                result.warnings.append(f"{condition.name}: {str(e)}")
                if condition.blocking:
                    result.passed = False
                    result.failed_conditions.append(condition.name)
        
        return result
    
    async def _check_with_timeout(
        self,
        check_func: callable,
        rule: Rule,
        timeout: int
    ) -> bool:
        """带超时的条件检查"""
        try:
            if asyncio.iscoroutinefunction(check_func):
                return await asyncio.wait_for(check_func(rule), timeout=timeout)
            else:
                return check_func(rule)
        except asyncio.TimeoutError:
            return False
```

```python
class RuleTransitionEventHandler:
    """状态转换事件处理器"""
    
    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.handlers: Dict[str, List[Callable]] = defaultdict(list)
        self._register_handlers()
    
    def _register_handlers(self):
        """注册内置处理器"""
        # 审计日志记录
        self.register(RuleTransitionEvents.TRANSITION_COMPLETE, 
                      self._log_transition)
        
        # 指标收集
        self.register(RuleTransitionEvents.POST_ACTIVATE,
                      self._record_activation_metric)
        self.register(RuleTransitionEvents.POST_DEPRECATE,
                      self._record_deprecation_metric)
        
        # 配额更新
        self.register(RuleTransitionEvents.POST_ACTIVATE,
                      self._update_quota_usage)
        self.register(RuleTransitionEvents.POST_PAUSE,
                      self._release_quota_usage)
        
        # 通知发送
        self.register(RuleTransitionEvents.POST_DEPRECATE,
                      self._send_deprecation_notification)
        
        # 缓存失效
        self.register(RuleTransitionEvents.TRANSITION_COMPLETE,
                      self._invalidate_rule_cache)
    
    def register(self, event_type: str, handler: Callable):
        """注册事件处理器"""
        self.handlers[event_type].append(handler)
    
    async def emit(
        self,
        event_type: str,
        event: StatusTransitionEvent
    ):
        """触发事件"""
        handlers = self.handlers.get(event_type, [])
        
        for handler in handlers:
            try:
                if asyncio.iscoroutinefunction(handler):
                    await handler(event)
                else:
                    handler(event)
            except Exception as e:
                logger.error(f"Event handler {handler.__name__} failed: {e}")
        
        # 发布到事件总线
        await self.event_bus.publish(event)
    
    # 内置处理器
    async def _log_transition(self, event: StatusTransitionEvent):
        """记录转换日志"""
        await self.audit_log.record({
            "type": "rule_status_transition",
            "rule_id": event.rule_id,
            "from": event.from_status.value,
            "to": event.to_status.value,
            "trigger": event.trigger,
            "triggered_by": event.triggered_by,
            "timestamp": event.triggered_at.isoformat()
        })
    
    async def _record_activation_metric(self, event: StatusTransitionEvent):
        """记录激活指标"""
        await self.metrics.increment("rules_activated_total")
        await self.metrics.gauge(
            "rules_by_status",
            {"status": "active"},
            await self._count_rules_by_status(RuleStatus.ACTIVE)
        )
    
    async def _update_quota_usage(self, event: StatusTransitionEvent):
        """更新配额使用"""
        rule = await self.rule_store.get(event.rule_id)
        await self.quota_manager.reserve(
            user_id=rule.user_id,
            resource="active_rules",
            amount=1
        )
    
    async def _send_deprecation_notification(self, event: StatusTransitionEvent):
        """发送废弃通知"""
        rule = await self.rule_store.get(event.rule_id)
        await self.notification_service.send(
            user_id=rule.user_id,
            template="rule_deprecated",
            params={
                "rule_name": rule.name,
                "deprecated_at": event.triggered_at.isoformat()
            }
        )
    
    async def _invalidate_rule_cache(self, event: StatusTransitionEvent):
        """失效规则缓存"""
        await self.cache.delete(f"rule:{event.rule_id}")
        await self.cache.delete(f"rule:{event.rule_id}:status")
```

#### 6.5.1 核心状态机类

```python
class RuleStateMachine:
    """规则状态机"""
    
    # 有效的状态转换映射
    TRANSITIONS = {
        (RuleStatus.DRAFT, RuleStatus.ACTIVE): "activate",
        (RuleStatus.ACTIVE, RuleStatus.PAUSED): "pause",
        (RuleStatus.ACTIVE, RuleStatus.DEPRECATED): "deprecate",
        (RuleStatus.PAUSED, RuleStatus.ACTIVE): "resume",
        (RuleStatus.PAUSED, RuleStatus.DEPRECATED): "deprecate",
        (RuleStatus.DEPRECATED, RuleStatus.DELETED): "delete",
        (RuleStatus.DELETED, None): None  # 终态
    }
    
    def __init__(
        self,
        rule: Rule,
        event_handler: RuleTransitionEventHandler,
        condition_checker: TransitionConditionChecker
    ):
        self.rule = rule
        self.event_handler = event_handler
        self.condition_checker = condition_checker
        self._transition_history: List[StatusTransitionEvent] = []
    
    @property
    def current_status(self) -> RuleStatus:
        """获取当前状态"""
        return self.rule.status
    
    def can_transition(self, target_status: RuleStatus) -> bool:
        """检查是否可以转换到目标状态"""
        key = (self.current_status, target_status)
        return key in self.TRANSITIONS
    
    async def transition(
        self,
        target_status: RuleStatus,
        trigger: str,
        trigger_by: str,
        reason: Optional[str] = None,
        context: Optional[Dict] = None
    ) -> TransitionResult:
        """执行状态转换"""
        # 1. 验证转换合法性
        if not self.can_transition(target_status):
            return TransitionResult(
                success=False,
                error=f"Invalid transition from {self.current_status} to {target_status}"
            )
        
        transition_name = self.TRANSITIONS[(self.current_status, target_status)]
        from_status = self.current_status
        
        # 2. 创建转换事件
        event = StatusTransitionEvent(
            event_id=self._generate_event_id(),
            rule_id=self.rule.id,
            from_status=from_status,
            to_status=target_status,
            trigger=trigger,
            trigger_by=trigger_by,
            triggered_at=datetime.now(),
            reason=reason,
            context=context or {},
            conditions_passed=[],
            conditions_failed=[]
        )
        
        # 3. 触发前置事件
        pre_event_type = f"pre_{transition_name}"
        await self.event_handler.emit(pre_event_type, event)
        
        # 4. 检查转换条件
        if transition_name:
            condition_result = await self.condition_checker.check_conditions(
                self.rule,
                transition_name
            )
            
            if not condition_result.passed:
                await self.event_handler.emit(
                    RuleTransitionEvents.TRANSITION_FAILED,
                    event
                )
                return TransitionResult(
                    success=False,
                    error="Transition conditions not met",
                    failed_conditions=condition_result.failed_conditions,
                    skipped_conditions=condition_result.skipped_conditions
                )
            
            event.conditions_passed = condition_result.conditions_passed
            event.conditions_failed = condition_result.conditions_failed
        
        # 5. 执行状态更新
        try:
            old_status = self.rule.status
            self.rule.status = target_status
            await self._save_rule()
            
            # 6. 记录转换历史
            self._transition_history.append(event)
            
            # 7. 触发后置事件
            post_event_type = f"post_{transition_name}"
            await self.event_handler.emit(post_event_type, event)
            
            # 8. 触发完成事件
            await self.event_handler.emit(
                RuleTransitionEvents.TRANSITION_COMPLETE,
                event
            )
            
            return TransitionResult(
                success=True,
                from_status=old_status,
                to_status=target_status,
                event=event
            )
            
        except Exception as e:
            # 转换失败，回滚状态
            self.rule.status = from_status
            await self.event_handler.emit(
                RuleTransitionEvents.TRANSITION_REVERTED,
                event
            )
            return TransitionResult(
                success=False,
                error=f"Transition failed: {str(e)}"
            )
```

#### 6.5.2 状态机装饰器

```python
def state_machine_transition(target_status: RuleStatus):
    """状态机转换装饰器"""
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(self, *args, **kwargs):
            # 获取状态机实例
            state_machine = getattr(self, 'state_machine', None)
            if not state_machine:
                return await func(self, *args, **kwargs)
            
            # 检查当前状态
            if not state_machine.can_transition(target_status):
                raise InvalidStateTransitionError(
                    f"Cannot transition from {state_machine.current_status} "
                    f"to {target_status}"
                )
            
            # 执行操作前检查
            await func(self, *args, **kwargs)
            
            # 执行状态转换
            trigger = kwargs.get('trigger', 'api')
            trigger_by = kwargs.get('trigger_by', 'system')
            reason = kwargs.get('reason')
            
            result = await state_machine.transition(
                target_status=target_status,
                trigger=trigger,
                trigger_by=trigger_by,
                reason=reason
            )
            
            if not result.success:
                raise StateTransitionError(result.error)
            
            return result
        
        return wrapper
    return decorator
```
